# 一、栈

```ts
class Stack {
  private items: number[] = []
  push(element: number) {
    this.items.push(element)
  }
  pop(): number {
    return this.items.pop()
  }
}
let stack = new Stack()
stack.push(1)
stack.push(2)
stack.pop()
```
